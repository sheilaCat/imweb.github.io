var $ = require(‘jquery’);

module.exports = {
data: {},
ready: function () {
var code = this.$.code,
dialog = this.$.dialog;
// 点击代码，弹出dialog
$(code.el).on(‘click’, function () {
dialog.show();
});
}
};
